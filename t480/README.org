** Wed Jan 14 05:16:16 PM WIB 2026

file: .config/etc/X11/99-disable-g2touch.conf

*** Disable permanently "G2Touch Multi-Touch by G2TSP":

#+begin_src bash :results value verbatim :exports both eval: never-export
  sudo vim /etc/X11/xorg.conf.d/99-disable-g2touch.conf
#+end_src

#+begin_src c :results value verbatim :exports both eval: never-export
  Section "InputClass"
      Identifier "Disable G2Touch"
      MatchProduct "G2Touch Multi-Touch by G2TSP"
      Option "Ignore" "on"
  EndSection
#+end_src

*** Blacklist permanently in Kernel module

file: .config/etc/modprobe.d/

#+begin_src bash :results value verbatim :exports both eval: never-export
  lsmod | grep -i g2touch
  # or
  dmesg | grep -i g2touch
#+end_src

If found, blacklist it:

#+begin_src bash :results value verbatim :exports both eval: never-export
  echo "blacklist g2touch" | sudo tee /etc/modprobe.d/blacklist-g2touch.conf
#+end_src

update initramfs:

#+begin_src bash :results value verbatim :exports both eval: never-export
